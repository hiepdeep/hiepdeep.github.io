Mảng sắp xếp JavaScript
Bằng cách kết hợp sort() và reverse(), bạn có thể sắp xếp một mảng theo thứ tự giảm dần
							
								const fruits = ["Banana", "Orange", "Apple", "Mango"];
								fruits.sort(); // Apple,Banana,Mango,Orange
								fruits.reverse(); // Mango,Apple,Orange,Banana
Hàm sort() sắp xếp các giá trị dưới dạng chuỗi
							
								const points = [40, 100, 1, 5, 25, 10];
								points.sort(function(a, b){return a - b}); // 1,5,10,25,40,100
								points.sort(function(a, b){return b - a}); // 100,40,25,10,5,1
Hàm sort() sắp xếp các đối tượng
							
								const cars = [
									{type: "Volvo", year: 2016},
									{type: "Saab", year: 2001},
									{type: "BMW", year: 2010}
								];
								cars.sort(function(a, b){return a.year - b.year});
								// Saab 2001
								// BMW 2010
								// Volvo 2016
Sử dụng Math.min.apply để tìm số nhỏ nhất trong một mảng
Hàm Math.min.apply(null, [1, 2, 3]) tương đương với Math.min(1, 2, 3)
							
								const points = [40, 100, 1, 5, 25, 10];
								function myArrayMin(arr) {
									return Math.min.apply(null, arr);
								}
								myArrayMin(points); // 1
Sử dụng Math.max.apply để tìm số lớn nhất trong một mảng
Hàm Math.max.apply(null, [1, 2, 3]) tương đương với Math.max(1, 2, 3)
							
								const points = [40, 100, 1, 5, 25, 10];
								function myArrayMax(arr) {
									return Math.max.apply(null, arr);
								}
								myArrayMax(points); // 100
Đối tượng toán học JavaScript
Math.round(x) trả về số nguyên gần nhất
							
								Math.round(4.6) // 5
								Math.round(4.5) // 5
								Math.round(4.4) // 4
Math.ceil(x) trả về giá trị của x được làm tròn lên số nguyên gần nhất
							
								Math.ceil(4.9); // 5
								Math.ceil(4.7); // 5
								Math.ceil(4.4); // 5
								Math.ceil(4.2); // 5
								Math.ceil(-4.2); // -4
Math.floor(x) trả về giá trị của x được làm tròn xuống thành số nguyên gần nhất
							
								Math.floor(4.9); // 4
								Math.floor(4.7); // 4
								Math.floor(4.4); // 4
								Math.floor(4.2); // 4
								Math.floor(-4.2); // -5
Math.trunc(x) trả về phần nguyên của x
							
								Math.trunc(4.9); // 4
								Math.trunc(4.7); // 4
								Math.trunc(4.4); // 4
								Math.trunc(4.2); // 4
								Math.trunc(-4.2); // -4
Math.sign(x) trả về phần nguyên của x
							
								Math.sign(4); // 1
								Math.sign(0); // 0
								Math.sign(-4); // -1
Math.abs(x) trả về giá trị tuyệt đối (dương) của x
							
								Math.abs(-4.7); // 4.7
Chuyển đổi values thành số và trả về giá trị lớn nhất
							
								Math.max(3, -5, 24) // 24
Chuyển đổi values thành số và trả về giá trị nhỏ nhất
							
								Math.min(3, -5, 24) // -5
Trả về một số nguyên ngẫu nhiên trong khoảng từ 0 đến 10

Hàm JavaScript này luôn trả về một số ngẫu nhiên nằm giữa min (đã bao gồm) và max (đã loại trừ)
							
								function getRndInteger(min, max) {
									return Math.floor(Math.random() * (max - min) ) + min;
								}
								getRndInteger(5, 10); // ramdom 5-9
Hàm JavaScript này luôn trả về một số ngẫu nhiên nằm giữa min và max (bao gồm cả hai)
							
								function getRndInteger(min, max) {
									return Math.floor(Math.random() * (max - min + 1) ) + min;
								}
								getRndInteger(5, 10); // ramdom 5-10

JavaScript Break and Continue
Câu lệnh break "nhảy ra" của một vòng lặp.
Trong ví dụ, câu lệnh break kết thúc vòng lặp ("ngắt" vòng lặp) khi bộ đếm vòng lặp (i) là 3.
							
								for (let i = 0; i < 10; i++) {
									if (i === 3) { break; }
									text += "The number is " + i + "<br>";
								}
								// 0 1 2
Câu lệnh continue "nhảy qua" một lần lặp trong vòng lặp.
Trong ví dụ, câu lệnh continue ngắt một lần lặp (trong vòng lặp), nếu một điều kiện xác định xảy ra và tiếp tục với lần lặp tiếp theo trong vòng lặp.
							
								for (let i = 0; i < 10; i++) {
									if (i === 3) { continue; }
									text += "The number is " + i + "<br>";
								}
								// 0 1 2 4 5 6 7 8 9
Phân tích cấu trúc JavaScript
Cú pháp gán cấu trúc giải nén các thuộc tính của đối tượng thành các biến
							
								let {firstName, lastName} = person;
Nó cũng có thể giải nén các mảng và bất kỳ đối tượng lặp nào khác
							
								let [firstName, lastName] = person;
Nó cũng có thể giải nén các mảng và bất kỳ đối tượng lặp nào khác
							
								const person = {
									firstName: "John",
									lastName: "Doe",
									age: 50
								};
								let {lastName, firstName} = person;
								document.getElementById("demo").innerHTML = firstName + " " + lastName; // John Doe
Đối với các thuộc tính có khả năng bị thiếu, chúng ta có thể đặt giá trị mặc định
							
								const person = {
									firstName: "John",
									lastName: "Doe",
									age: 50
								};
								let {firstName, lastName, country = "US"} = person;
								document.getElementById("demo").innerHTML = firstName + " " + lastName + " " + country; // John Doe US
Có thể kết thúc cú pháp phân rã bằng thuộc tính rest
							
								const numbers = [10, 20, 30, 40, 50, 60, 70];
								const [a,b, ...rest] = numbers;
								document.getElementById("demo").innerHTML = a; // 10
								document.getElementById("demo").innerHTML = b; // 20
								document.getElementById("demo").innerHTML = rest; // 30,40,50,60,70
Hoặc
							
								const numbers = [10, 20, 30, 40, 50, 60, 70];
								const [...rest] = numbers;
								document.getElementById("demo").innerHTML = rest; // 10,20,30,40,50,60,70
Bản đồ phân hủy cấu trúc
							
								const fruits = new Map([
									["apples", 500],
									["bananas", 300],
									["oranges", 200]
								]);
								let text = "";
								for (const [key, value] of fruits) {
